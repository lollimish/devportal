participant "<b>WebRTC App" as browser
participant "<b>APIGW" as apig

User is registered and 
does have a <b>session ID</b>
end note
group start a call
browser->apig : POST /sessions/{sessionId}/calls(sdp)
apig->browser :HTTP 201 Created(callId)
end

group Event Channel loop  
browser->apig : GET /sessions/{sessionId}/events 
apig->browser : 200 OK (Events Received)
end
